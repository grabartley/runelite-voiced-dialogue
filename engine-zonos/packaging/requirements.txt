# Runtime dependencies for the self-contained Zonos GPU engine bundle.
#
# These are installed into the build venv before PyInstaller freezes the engine. The PyTorch CUDA
# wheels are pulled from the CUDA index URL (see build_bundle.py / the release workflow, which pass
# --extra-index-url https://download.pytorch.org/whl/cu124). Those wheels carry their own bundled
# CUDA runtime, so the END USER needs only an NVIDIA driver: no CUDA toolkit, no system Python.
#
# Versions are pinned for reproducible bundles. Bump deliberately and re-run --selftest on a GPU box
# before publishing.

# Deep-learning runtime (CUDA build resolved via the cu124 extra index in the build step).
# MUST satisfy Zonos's own dependency floor (Zonos main requires torch>=2.5.1 / torchaudio>=2.5.1).
# When this was pinned to 2.4.1, the Zonos git install upgraded torch to satisfy its >=2.5.1 floor and
# -- because that install lacked the cu124 index -- pulled the CPU wheel (torch 2.12.1+cpu), shipping a
# bundle that reported no usable GPU on every box. 2.5.1+cu124 has cp312 wheels and satisfies Zonos.
torch==2.5.1
torchaudio==2.5.1

# Pin numpy below 2.4. numpy 2.4 added a guard that raises
# "cannot load module more than once per process" when its C extension is initialized twice, which
# the frozen PyInstaller bootstrap does (it removes numpy in cleanup_loaded_modules, then torch
# re-imports it). Unpinned, the build pulled numpy >=2.4 and the GPU probe failed with that exact
# error and reported no GPU on every box (the failure was misattributed to torch in #77). The range
# [2.2.2, 2.4) satisfies both Zonos (numpy>=2.2.2) and torch 2.5.1, so this stays compatible. Bump
# deliberately once numpy/PyInstaller resolve the double-load, and re-test --selftest on a GPU box.
numpy<2.4

# transformers 5.x resolves submodules by scanning the filesystem (create_import_structure_from_path
# -> os.scandir), which fails in a frozen PyInstaller bundle where modules live in the archive, not on
# disk: synthesis died with FileNotFoundError on transformers/models/__init__.pyc. 4.x uses static
# _import_structure dicts and freezes correctly. Zonos requires >=4.48.1.
transformers==4.48.1

# Zonos-v0.1 (Apache-2.0) and its synthesis dependencies. Installed from the upstream repo at a
# pinned commit by build_bundle.py so the model API the engine calls (make_cond_dict,
# prepare_conditioning, generate, autoencoder.decode, make_speaker_embedding) is stable.
# (Zonos is not on PyPI; the build step installs it from git — see packaging/README intent in
# build_bundle.py.)

# Phonemizer backend Zonos uses for text normalization.
phonemizer==3.3.0

# Bundles the espeak-ng shared library + data per-OS (including Windows), so a user box needs no
# system espeak-ng install. The engine points phonemizer at it (see engine._configure_espeak).
espeakng-loader==0.2.4

# PyInstaller is a build-time tool, not a runtime dep, but pinned here so the build env is one file.
pyinstaller==6.10.0
